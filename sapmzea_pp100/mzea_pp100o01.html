MODULE FILL_DYNP_0100 OUTPUT.

  IF ZEA_T001W-PNAME1 IS INITIAL.
    CLEAR ZEA_T001W-WERKS.
  ELSE.
    SELECT SINGLE WERKS
      FROM ZEA_T001W
      INTO ZEA_T001W-WERKS
     WHERE PNAME1 EQ ZEA_T001W-PNAME1.
  ENDIF.

ENDMODULE.
